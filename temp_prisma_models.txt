> model PaymentDefinition {
    id                   Int             @id @default(autoincrement())
    title                String
    calculationType      CalculationType // FIXED | PER_UNIT | PERCENTAGE | FORMULA
    amount               Decimal? // For FIXED / PERCENTAGE
    unitName             String? // e.g., 'sq_ft'
    unitRate             Decimal? // for PER_UNIT
    isRecurring          Boolean
    recurrenceFrequency  RecurrenceType? // MONTHLY, YEARLY, etc.
    recurrenceCount      Int? // Optional
    taxPercent           Decimal? // e.g., 18
    isInterestApplicable Boolean
    interestType         InterestType? // SIMPLE, COMPOUND
    interestRate         Decimal?
    enableLateFee        Boolean         @default(false)

    effectiveFrom DateTime   @default(now())
    effectiveTo   DateTime?
    status        StatusType @default(ACTIVE)
    createdAt     DateTime   @default(now())
    updatedAt     DateTime   @updatedAt
  
    form_id Int

    lateFeeRules LateFeeRule[]
    invoices     PaymentInvoice[]
    demandModifications  DemandModification[] @relation("DefToMods")
