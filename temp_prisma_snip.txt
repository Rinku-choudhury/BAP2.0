// Master: holds definition for each metadata category
model Master {
  id          Int      @id @default(autoincrement()) // unique identifier for the category
  name        String // e.g. "Scheme", "Output"

}

  status    Boolean  @default(true) // active/inactive flag
  createdOn DateTime @default(now()) // timestamp of creation

  // Relations
  master           Master          @relation(fields: [masterId], references: [id])
  primaryHeaders   MappingHeader[] @relation("PrimaryRecord")
  secondaryHeaders MappingHeader[] @relation("SecondaryRecord")
  detailMappings   MappingDetail[]
}
